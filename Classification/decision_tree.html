def calculateMetrix(name, trueValues, predictValues):

    print(name)
    print('Confusion_matrix:')
    print(pd.DataFrame(confusion_matrix(trueValues, predictValues), index=model.classes_, columns=model.classes_))

    print('Accuracy_score:')
    print(accuracy_score(trueValues, predictValues, normalize=False))
    print('Accuracy_score_normalize:')
    print(accuracy_score(trueValues, predictValues, normalize=True))

    print('Precision_score:')
    print(precision_score(trueValues, predictValues, average=None))

    print('Recall_score:')
    print(recall_score(trueValues, predictValues,average=None))

    print('Classification_report:')
    print(classification_report(trueValues, predictValues))
